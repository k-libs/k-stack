class Stack<T>

Generic Stack

A simple stack implementation.

Example:

// Create a new stack
val stack = Stack<Int>()

// Push 3, 2, then 1 onto the stack.
for (i in 3 downTo 1)
  stack.push(i)

// Pop 1, 2, then 3 from the stack.
for (i in 1 .. 3)
  require(stack.pop() == i)
